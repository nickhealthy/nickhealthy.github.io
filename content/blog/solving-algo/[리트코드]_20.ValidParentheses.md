```python
def isValid(s):
    stack = []
    table = {
        ')' : '(',
        '}' : '{',
        ']' : '[',
    }
    # ìŠ¤íƒ ì´ìš© ì˜ˆì™¸ ì²˜ë¦¬ ë° ì¼ì¹˜ ì—¬ë¶€ íŒë³„
    for char in s:
        if char not in table:
            stack.append(char)
        elif not stack or table[char] != stack.pop():
            return False


    return len(stack) == 0

print(isValid('()[]{}'))
```
