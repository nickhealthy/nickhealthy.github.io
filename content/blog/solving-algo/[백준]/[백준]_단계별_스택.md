```python
import sys
n = int(sys.stdin.readline())

for i in range(n):
    input = sys.stdin.readline().rstrip()
    stack = [0]
    for j in input:
        if j == "(":
            stack.append(j)
        else:
            try:
                stack.pop()
            except:
                continue

    if len(stack) == 1:
        if stack[0] == "(":
            print("NO")
        else:
            print("YES")
    else:
        print("NO")
```

```python
import sys
input = sys.stdin.readline
while 1:
    string = input().rstrip()
    stack = []
    true_flag = 1
    for cha in string:
        if cha == '(' or cha == '[':
            stack.append(cha)
        elif cha == ')':
            if stack and stack[-1] == '(':
                stack.pop()
            else:
                true_flag = 0
                break
        elif cha == ']':
            if stack and stack[-1] == '[':
                stack.pop()
            else:
                true_flag = 0
                break
    if string == '.':
        break
    print("yes" if true_flag and not(stack) else "no")
```
