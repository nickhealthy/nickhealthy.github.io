```python
import sys
input = sys.stdin.readline()

n = int(input)

def fibo(n):
    result = []
    first = 1
    second = 1
    if n > 1:
        result.append(first)
    result.append(second)
    for _ in range(2, n):
        third = first + second
        first = second
        second = third
        result.append(third)
    return result.pop()


print(fibo(n))
```
