class Solution:
    def longestPalindrome(self, s: str) -> str:
        # íŒë³„ ëŒ€ìƒì´ ì•„ë‹Œ ê²ƒì€ ë¹ ë¥´ê²Œ return
        if len(s) < 2 or s == s[::-1]:
            return s

        # íˆ¬ í¬ì¸í„°ë¥¼ ì´ìš©í•´ í™•ì¥
        def expand(left, right):
            while left >= 0 and right <= len(s) and s[left] == s[right - 1]:
                left -= 1
                right += 1
            return s[left + 1 : right - 1]

        # í™€ìˆ˜ì™€ ì§ìˆ˜ë¡œ íŒ°ë¦°ë“œë¡¬ì„ ëª¨ë‘ íŒë³„

        for i in range(len(s) - 1):
            result = max(result,
                            expand(i, i + 1),
                            expand(i, i + 2),
                            key=len)
        return result
